```js
// prices: [50, 120, 80, 200, 60, 150] のような6要素配列（50〜200の整数）
function randomPrices() {
  return Array.from({length: 6}, () => Math.floor(Math.random() * 151) + 50);
}

// パターンA（定量）: 初月の価格で個数を決め、毎月同じ個数を買う
function calcA(monthly, prices) {
  const qty = Math.round(monthly / prices[0]);
  const total = qty * 6;
  const spend = prices.reduce((s, p) => s + qty * p, 0);
  const avg = Math.round(spend / total);
  return { qty, total, spend, avg, counts: Array(6).fill(qty) };
}

// パターンB（定額）: 毎月同じ金額で買えるだけ買う
function calcB(monthly, prices) {
  const counts = prices.map(p => Math.floor(monthly / p));
  const total = counts.reduce((s, n) => s + n, 0);
  const spend = monthly * 6;
  const avg = Math.round(spend / total);
  return { total, spend, avg, counts };
}

// リンゴHTML（上限15個、超過分は +N 表示）
function appleHtml(n) {
  const MAX = 15;
  const shown = Math.min(n, MAX);
  const extra = n - MAX;
  return '🍎'.repeat(shown) + (extra > 0 ? `<span style="font-size:11px;color:#888">+${extra}</span>` : '');
}

// 平均価格との比較でバッジ色を決める
function priceBadge(price, avg) {
  const cheap = price <= avg;
  const bg = cheap ? '#1D9E75' : '#E07B54';
  return `<span style="display:inline-block;background:${bg};color:#fff;border-radius:4px;padding:1px 6px;font-size:11px;font-weight:600">${price}円</span>`;
}
```

```html
let monthly = 30000;
let prices = [];
let calcAResult = null;
let calcBResult = null;

document.querySelectorAll('.amt-btn').forEach(b => {
  b.onclick = () => {
    document.querySelectorAll('.amt-btn').forEach(x => x.classList.remove('on'));
    b.classList.add('on');
    monthly = +b.dataset.v;

function randomPrices() {
  return Array.from({length: 6}, () => Math.floor(Math.random() * 151) + 50);
}

function calcA(monthly, prices) {
  const qty = Math.round(monthly / prices[0]);
  const total = qty * 6;
  const spend = prices.reduce((s, p) => s + qty * p, 0);
  const avg = Math.round(spend / total);
  return { qty, total, spend, avg, counts: Array(6).fill(qty) };
}

function calcB(monthly, prices) {
  const counts = prices.map(p => Math.floor(monthly / p));
  const total = counts.reduce((s, n) => s + n, 0);
  const spend = monthly * 6;
  const avg = Math.round(spend / total);
  return { total, spend, avg, counts };
}

function appleHtml(n) {
  const MAX = 15;
  const shown = Math.min(n, MAX);
  const extra = n - MAX;
  return '🍎'.repeat(shown) + (extra > 0 ? `<span style="font-size:11px;color:#888">+${extra}</span>` : '');
}

function priceBadge(price, avg) {
  const cheap = price <= avg;
  const bg = cheap ? '#1D9E75' : '#E07B54';
  return `<span style="display:inline-block;background:${bg};color:#fff;border-radius:4px;padding:1px 6px;font-size:11px;font-weight:600">${price}円</span>`;
}

const MONTH_NAMES = ['1月','2月','3月','4月','5月','6月'];

function renderMonths(containerId, counts, avgPrice) {
  const container = document.getElementById(containerId);
  container.innerHTML = MONTH_NAMES.map((m, i) => `
    <div class="month-col">
      <div class="month-apples">${appleHtml(counts[i])}</div>
      ${priceBadge(prices[i], avgPrice)}
      <div class="month-name">${m}</div>
    </div>
  `).join('');
}

function renderPricesOnly() {
  ['months-a','months-b'].forEach(id => {
    const container = document.getElementById(id);
    container.innerHTML = MONTH_NAMES.map((m, i) => {
      const avgPrice = Math.round(prices.reduce((s,p)=>s+p,0)/prices.length);
      return `<div class="month-col">
        <div class="month-apples" style="min-height:50px;"></div>
        ${priceBadge(prices[i], avgPrice)}
        <div class="month-name">${m}</div>
      </div>`;
    }).join('');
  });
}

function setStepDone(n) {
  document.getElementById('sn'+n).classList.add('done');
  document.getElementById('sb'+n).classList.remove('active');
  document.getElementById('sb'+n).classList.add('done');
  document.getElementById('sb'+n).disabled = true;
}

function activateStep(n) {
  document.getElementById('sb'+n).disabled = false;
  document.getElementById('sb'+n).classList.add('active');
}

function doStep1() {
  prices = randomPrices();
  calcAResult = calcA(monthly, prices);
  calcBResult = calcB(monthly, prices);
  renderPricesOnly();
  document.getElementById('box-a').classList.add('visible');
  document.getElementById('box-b').classList.add('visible');
  setStepDone(1);
  activateStep(2);
}

function doStep2() {
  const avgPrice = Math.round(prices.reduce((s,p)=>s+p,0)/prices.length);
  renderMonths('months-a', calcAResult.counts, avgPrice);
  document.getElementById('summary-a').textContent =
    `合計 ${calcAResult.total}個 ／ 平均単価 ${calcAResult.avg}円`;
  setStepDone(2);
  activateStep(3);
}

function doStep3() {
  const avgPrice = Math.round(prices.reduce((s,p)=>s+p,0)/prices.length);
  renderMonths('months-b', calcBResult.counts, avgPrice);
  document.getElementById('summary-b').textContent =
    `合計 ${calcBResult.total}個 ／ 平均単価 ${calcBResult.avg}円`;
  setStepDone(3);
  activateStep(4);

function doStep4() {
  document.getElementById('r-total-a').textContent = `${calcAResult.total}個`;
  document.getElementById('r-total-b').textContent = `${calcBResult.total}個`;
  document.getElementById('r-avg-a').textContent = `${calcAResult.avg}円`;
  document.getElementById('r-avg-b').textContent = `${calcBResult.avg}円`;

  const diff = calcBResult.total - calcAResult.total;
  const avgDiff = calcAResult.avg - calcBResult.avg;
```
